func search(nums []int, target int) int {
    l, r := 0, len(nums)-1
    for l <= r {
        mid := l + (r-l)/2
        if nums[mid] == target {
            return mid
        }

        if nums[mid] >= nums[l] {
            if nums[mid] > target && target >= nums[l] {
                r = mid - 1
            } else {
                l = mid + 1
            }
        } else {
            if nums[mid] < target && target <= nums[r] {
                l = mid + 1
            } else {
                r = mid - 1
            }
        }
    }
    return -1
}

func findMin(nums []int) int {
    l, r := 0, len(nums) -1
    for l < r {
        mid := l + (r - l) / 2
        if nums[mid] < nums[r] {
            r = mid
        } else {
            l = mid + 1
        }
    }
    return nums[l]
}
